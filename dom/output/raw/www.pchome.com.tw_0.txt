    var delay = delay||1000,speed = speed||20,h = h||20;

            if(slideBox.scrollTop%h == 0){
                clearInterval(tid);
                slideBox.appendChild(slideBox.getElementsByTagName(stf)[0]);
                slideBox.scrollTop = 0;
                setTimeout(s, delay);
            }

    slideBox.onmouseout=function(){pause=false;}
    setTimeout(s, delay);
