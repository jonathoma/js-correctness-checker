                                                /**
* Tirage au sort d'utilisateur pour la mise en place d'un système de déploiement progressif
* @param {object}[] rate objet de définiton des différents taux
* @param {string}[] rate[].value Valeur du cookie quand l'utilisateur n'est pas sélectionné
* @param {integer}[] rate[].selectedRate Pourcentage de chance de sélection (entre 1 et 100), précision jusqu'à 4 décimales
* @param {integer} currentRound Numéro du tour en cours
* @param {string} [cookieName="o-lottery"] Nom du cookie
* @param {integer} [expireDays=1] Durée d'expiration du cookie en jour
* @returns {boolean} True si le tirage au sort à réussi, false sinon.
*/
function o_progressiveDeploymentCustom(rate, currentRound, cookieName, expireDays) {
    if (rate == null) {
        return false;
    }
 
    var round = currentRound-1 || 0,
        cookie = cookieName || "o-lottery",
        expire = expireDays || 1,
        retryRate,
        result,
        i;
 
    //Seules les 4 premières décimales du taux sont conservées.
    //Pour simplifier les traitements, les décimales sont retirées en multipliant par 10000
    //Dans la suite du code, tous les pourcentages seront multipliés par 1000000 au lieu de 100
    for (i in rate) {
        if (rate.hasOwnProperty(i)) {
            rate[i].selectedRate = Math.round(rate[i].selectedRate * 10000);
        }
    }
 
    //Calcul du taux de repêchage des personnes non sélectionnées aux tours précédents
    //Le calcul se fait en divisant l'augmentation du pourcentage de sélection par le pourcentage de non
    //sélectionné au tour précédent
    //Ex : tour 1 : 0: 75%, 1: 25%
    //     tour 2 : 0: 50%, 1: 50%
    //Le taux de repêchage est de (50-25)/75 -> 33%
    if (round > 0) {
        retryRate = Math.round(((rate[round].selectedRate - rate[round-1].selectedRate) / (1000000 - rate[round-1].selectedRate)) * 1000000, 2);
    }
 
    //Si la personne n'a pas déjà été sélectionnée
    if (o_sGetCookie(cookie) !== "1") {
        var random = parseInt((Math.random() * 1000000) + 1);
 
        //Si c'est une nouvelle personne
        if (o_sGetCookie(cookie) === false) {
            result = (random <= rate[round].selectedRate) ? "1" : rate[round].value;
            o_bSetCookie(cookie, result, "domain = .orange.fr, expires=" + expire);
        }
 
        //Si c'est une personne n'ayant pas été sélectionnée lors d'un tour précédent
        else if (o_sGetCookie(cookie) !== rate[round].value) {
            result = (random <= retryRate) ? "1" : rate[round].value;
            o_bSetCookie(cookie, result, "domain = .orange.fr, expires=" + expire);
        }
 
    }
 
    return true;
 
}
