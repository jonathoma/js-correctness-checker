//ページ先頭の横並びのバナー(1つずつ)を戻り値とする
function createSubBanner(ad, width, height)
{
    if (typeof(width) == "undefined") {
        //引数width指定なしの場合
        width = 152;
    }

    if (typeof(height) == "undefined") {
        //引数height指定なしの場合
        height = 64;
    }

    var img = document.createElement("img");
    img.height = height;
    img.width = width;

    var link = document.createElement("a");
    link.href = ad.clickUrl;

    if (ad.creative.images && ad.creative.images[0]) {
        //コンボクリエイティブの場合
        if (ad.creative.content) {
            for (contentIndex in ad.creative.content) {
                //オプションが設定されている場合
                switch (ad.creative.content[contentIndex].key) {
                case "target":
                    //管理画面でtargetの値を登録しないとcontent属性値が空配列になることに注意！
                    link.target = ad.creative.content[contentIndex].value;
                    break;
                case "説明":
                    //説明が登録されていたら代替テキストをセット
                    img.alt = ad.creative.content[contentIndex].value;
                    break;
                default:
                }
            }
        }
        img.src = ad.creative.images[0].source;
    } else {
        //イメージクリエイティブの場合
        img.src = ad.creative.source;

        if (ad.creative.description) {
            //説明が登録されていたら代替テキストをセット
            img.alt = ad.creative.description;
        }
    }

    link.appendChild(img);
    return link;
}

function renderSubAd(data, context) {
    if (data) {
        if (isHeaderSubAd) return;
        //広告スペースが存在する場合は広告オブジェクトを取得
        var ads = data;

        //左側のバナー
        if (ads[0]) {
            $("ul.box-service-pickup > li:eq(0)").append(createSubBanner(ads[0], 152, 64));
        }

        //右側のバナー
        if (ads[1]) {
            $("ul.box-service-pickup > li:eq(1)").append(createSubBanner(ads[1], 152, 64));
        }

        if (ads.length < 2) {
            //バナー数が不足している場合
            insertAlternateHeaderSubAd();
            return;
        }
        isHeaderSubAd = true;
    }
}
